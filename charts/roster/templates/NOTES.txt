   kubectl -n {{ .Release.Namespace }} exec deployment/{{ include "roster.fullname" . }} -- roster status

   kubectl -n {{ .Release.Namespace }} exec deployment/{{ include "roster.fullname" . }} -- roster logs {{ .Values.repo }} -f
